00145     # ctypes data types with the same value don't compare as equal (e.g.,

00150     def _values(self, obj):
00151         values = []
00152         for f in obj._fields_:
00153             v = getattr(obj, f[0])
00154             # If we look like a pointer type, just worry about the address
00155             # to which we are pointing, otherwise try to obtain the ctypes
00156             # data type value.  Note that ctypes can raise ValueError if we
00157             # accidentally try to access faulty pointers, so just safety
00158             # block the whole process.
00159             try:
00160                 if isinstance(v, ctypes._Pointer):
00161                     # We'd like two NULL pointers to be equal, but ctypes will
00162                     # fail obtaining the address, so use None for such pointers
00163                     if v:
00164                         v = addressof(v.contents)
00165                     else:
00166                         v = None
00167                 elif isinstance(v, ctypes._SimpleCData):
00168                     v = v.value
00169             except Exception:
00170                 pass
00171             values.append(v)

00680 def check_slice(sl, length):
00681     if isinstance(sl, slice):
00682         if sl.start is None:
00683             start = 0
00684         else:
00685             start = sl.start
00686         if not 0 <= start < length:
00687             raise IndexError("Item %d is not in range [0,%d)" % (start, length))
00688 
00689         if sl.step is None:
00690             step = 1
00691         else:
00692             step = sl.step
00693             
00694         if step >= 0:
00695             stop = length
00696         else:
00697             stop = -1
00698         if sl.stop is not None:
00699             if step >= 0:
00700                 if stop > sl.stop:
00701                     stop = sl.stop
00702                 if stop < start:
00703                     stop = start
00704             else:
00705                 if stop < sl.stop:
00706                     stop = sl.stop
00707                 if stop > start:
00708                     stop = start
00709     else:
00710         sl = int(sl)
00711         if not 0 <= sl < length:
00712             raise IndexError("Item %d is not in [0,%d)" % (sl, length))
00713         start = sl
00714         stop = sl+1
00715         step = 1
00716         
00717     return slice(start, stop, step)

00761     def get_pixel_or_slice2d(self, key):
