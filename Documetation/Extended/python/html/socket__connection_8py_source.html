00093         if url.lower().startswith("socket://"): url = url[9:]

00101                     if '=' in option:

00105                     if option == 'logging':
00106                         logging.basicConfig()   # XXX is that good to call it here?
00107                         self.logger = logging.getLogger('pySerial.socket')
00108                         self.logger.setLevel(LOGGER_LEVELS[value])
00109                         self.logger.debug('enabled logging')
00110                     else:
00111                         raise ValueError('unknown option: %r' % (option,))

00113             host, port = url.split(':', 1) # may raise ValueError because of unpacking
