00434     # select based implementation, proved to work on many systems
00435     def read(self, size=1):
00436         """Read size bytes from the serial port. If a timeout is set it may
00437            return less characters as requested. With no timeout it will block
00438            until the requested number of bytes is read."""
00439         if self.fd is None: raise portNotOpenError
00440         read = bytearray()
00441         while len(read) < size:
00442             ready,_,_ = select.select([self.fd],[],[], self._timeout)
00443             # If select was used with a timeout, and the timeout occurs, it
00444             # returns with empty lists -> thus abort read operation.
00445             # For timeout == 0 (non-blocking operation) also abort when there
00446             # is nothing to read.
00447             if not ready:
00448                 break   # timeout
00449             buf = os.read(self.fd, size-len(read))
00450             # read should always return some data as select reported it was
00451             # ready to read when we get to this point.
00452             if not buf:
00453                 # Disconnected devices, at least on Linux, show the
00454                 # behavior that they are always ready to read immediately
00455                 # but reading returns nothing.
00456                 raise SerialException('device reports readiness to read but returned no data (device disconnected?)')
00457             read.extend(buf)
00458         return bytes(read)

00460     def write(self, data):
00461         """Output the given string over the serial port."""
00462         if self.fd is None: raise portNotOpenError
00463         t = len(data)
00464         d = data
00465         if self._writeTimeout is not None and self._writeTimeout > 0:
00466             timeout = time.time() + self._writeTimeout
00467         else:
00468             timeout = None
00469         while t > 0:
00470             try:
00471                 n = os.write(self.fd, d)
00472                 if timeout:
00473                     # when timeout is set, use select to wait for being ready
00474                     # with the time left as timeout
00475                     timeleft = timeout - time.time()
00476                     if timeleft < 0:
00477                         raise writeTimeoutError
00478                     _, ready, _ = select.select([], [self.fd], [], timeleft)
00479                     if not ready:
00480                         raise writeTimeoutError

00482                 t = t - n
00483             except OSError, v:
00484                 if v.errno != errno.EAGAIN:
00485                     raise SerialException('write failed: %s' % (v,))

00607     def read(self, size=1):
00608         """Read size bytes from the serial port. If a timeout is set it may
00609            return less characters as requested. With no timeout it will block
00610            until the requested number of bytes is read."""
00611         if self.fd is None: raise portNotOpenError
00612         read = bytearray()
00613         poll = select.poll()
00614         poll.register(self.fd, select.POLLIN|select.POLLERR|select.POLLHUP|select.POLLNVAL)
00615         if size > 0:
00616             while len(read) < size:
00617                 # print "\tread(): size",size, "have", len(read)    #debug
00618                 # wait until device becomes ready to read (or something fails)
00619                 for fd, event in poll.poll(self._timeout*1000):
00620                     if event & (select.POLLERR|select.POLLHUP|select.POLLNVAL):
00621                         raise SerialException('device reports error (poll)')
00622                     #  we don't care if it is select.POLLIN or timeout, that's
00623                     #  handled below
00624                 buf = os.read(self.fd, size - len(read))
00625                 read.extend(buf)
00626                 if ((self._timeout is not None and self._timeout >= 0) or 
00627                     (self._interCharTimeout is not None and self._interCharTimeout > 0)) and not buf:
00628                     break   # early abort on timeout
00629         return bytes(read)
