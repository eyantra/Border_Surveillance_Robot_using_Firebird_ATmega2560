00115     def read(self, size=1):
00116         """Read size bytes from the serial port. If a timeout is set it may
00117         return less characters as requested. With no timeout it will block
00118         until the requested number of bytes is read."""
00119         if not self._isOpen: raise portNotOpenError
00120         if self._timeout is not None:
00121             timeout = time.time() + self._timeout
00122         else:
00123             timeout = None
00124         data = bytearray()
00125         while len(data) < size:
00126             self.buffer_lock.acquire()
00127             try:
00128                 block = to_bytes(self.loop_buffer[:size])
00129                 del self.loop_buffer[:size]
00130             finally:
00131                 self.buffer_lock.release()
00132             data += block
00133             # check for timeout now, after data has been read.
00134             # useful for timeout = 0 (non blocking) read
00135             if timeout and time.time() > timeout:
00136                 break
00137         return bytes(data)

00139     def write(self, data):
00140         """Output the given string over the serial port. Can block if the
00141         connection is blocked. May raise SerialException if the connection is
00142         closed."""
00143         if not self._isOpen: raise portNotOpenError
00144         # calculate aprox time that would be used to send the data
00145         time_used_to_send = 10.0*len(data) / self._baudrate
00146         # when a write timeout is configured check if we would be successful
00147         # (not sending anything, not even the part that would have time)
00148         if self._writeTimeout is not None and time_used_to_send > self._writeTimeout:
00149             time.sleep(self._writeTimeout) # must wait so that unit test succeeds
00150             raise writeTimeoutError
00151         self.buffer_lock.acquire()
00152         try:
00153             self.loop_buffer += bytes(data)
00154         finally:
00155             self.buffer_lock.release()
