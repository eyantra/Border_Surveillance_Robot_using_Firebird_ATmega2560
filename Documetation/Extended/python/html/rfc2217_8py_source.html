00049 # future, so lets use an URL scheme.
00050 # for RFC2217 compliant servers we will use this:
00051 #    rfc2217://<host>:<port>[/option[/option...]]

00053 # options:
00054 # - "debug" print diagnostic messages
00055 # - "ign_set_control": do not look at the answers to SET_CONTROL

00260                 self.active = True
00261                 if self.activation_callback is not None:
00262                     self.activation_callback()

00264                 pass
00265             elif self.state is INACTIVE:
00266                 self.state = ACTIVE

00268                 self.active = True

00272                 self.connection.telnetSendOption(self.send_no, self.option)

00276             if self.state is REQUESTED:
00277                 self.state = INACTIVE
00278                 self.active = False
00279             elif self.state is ACTIVE:

00942     def check_modem_lines(self, force_notification=False):
00943         modemstate = (
00944             (self.serial.getCTS() and MODEMSTATE_MASK_CTS) |
00945             (self.serial.getDSR() and MODEMSTATE_MASK_DSR) |
00946             (self.serial.getRI() and MODEMSTATE_MASK_RI) |
00947             (self.serial.getCD() and MODEMSTATE_MASK_CD)

00951         if deltas & MODEMSTATE_MASK_CTS:
00952             modemstate |= MODEMSTATE_MASK_CTS_CHANGE
00953         if deltas & MODEMSTATE_MASK_DSR:
00954             modemstate |= MODEMSTATE_MASK_DSR_CHANGE
00955         if deltas & MODEMSTATE_MASK_RI:
00956             modemstate |= MODEMSTATE_MASK_RI_CHANGE
00957         if deltas & MODEMSTATE_MASK_CD:
00958             modemstate |= MODEMSTATE_MASK_CD_CHANGE

00960         # notification. suppress notifications when client is not rfc2217
00961         if modemstate != self.last_modemstate or force_notification:
00962             if (self._client_is_rfc2217 and (modemstate & self.modemstate_mask)) or force_notification:
00963                 self.rfc2217SendSubnegotiation(
