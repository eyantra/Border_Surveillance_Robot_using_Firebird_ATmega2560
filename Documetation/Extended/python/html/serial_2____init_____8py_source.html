00028 def serial_for_url(url, *args, **kwargs):
00029     """Get a native, a RFC2217 or socket implementation of the Serial class,
00030     depending on port/url. The port is not opened when the keyword parameter
00031     'do_not_open' is true, by default it is."""
00032     # check remove extra parameter to not confuse the Serial class
00033     do_open = 'do_not_open' not in kwargs or not kwargs['do_not_open']
00034     if 'do_not_open' in kwargs: del kwargs['do_not_open']
00035     # the default is to use the native version
00036     klass = Serial   # 'native' implementation
00037     # check port type and get class
00038     try:
00039         url_nocase = url.lower()
00040     except AttributeError:
00041         # its not a string, use default
00042         pass
00043     else:
00044         if url_nocase.startswith('rfc2217://'):
00045             import rfc2217  # late import, so that users that don't use it don't have to load it
00046             klass = rfc2217.Serial # RFC2217 implementation
00047         elif url_nocase.startswith('socket://'):
00048             import socket_connection  # late import, so that users that don't use it don't have to load it
00049             klass = socket_connection.Serial
00050         elif url_nocase.startswith('loop://'):
00051             import loopback_connection  # late import, so that users that don't use it don't have to load it
00052             klass = loopback_connection.Serial
00053         else:
00054             klass = Serial   # 'native' implementation
00055     # instantiate and open when desired
00056     instance = klass(None, *args, **kwargs)
00057     instance.port = url
00058     if do_open:
00059         instance.open()
